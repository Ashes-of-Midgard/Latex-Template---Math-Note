% 边框默认格式

% 定义
\newcounter{defnCounter}[section]
\setcounter{defnCounter}{0}
\newenvironment{defn}[1][]
{
    \refstepcounter{defnCounter}
    \begin{tcolorbox}[colback=cyan!10!white,colframe=black!30!white]
    \textbf{定义\thesection.\thedefnCounter #1：}
}
{
    \end{tcolorbox}
}

% 定理、引理、推论、命题和准则
\newcounter{thmCounter}[section]
\setcounter{thmCounter}{0}
\newenvironment{thm}[1][]
{
    \refstepcounter{thmCounter}
    \begin{tcolorbox}[colback=yellow!10!white,colframe=black!30!white]
    \textbf{定理\thesection.\thethmCounter #1：}
}
{
    \end{tcolorbox}
}

\newenvironment{lemma}[1][]
{
    \refstepcounter{thmCounter}
    \begin{tcolorbox}[colback=black!3!white,colframe=black!30!white]
    \textbf{引理\thesection.\thethmCounter #1：}
}
{
    \end{tcolorbox}
}

\newenvironment{corollary}[1][]
{
    \refstepcounter{thmCounter}
    \begin{tcolorbox}[colback=black!3!white,colframe=black!30!white]
    \textbf{推论\thesection.\thethmCounter #1：}
}
{
    \end{tcolorbox}
}

\newenvironment{criterion}[1][]
{
    \refstepcounter{thmCounter}
    \begin{tcolorbox}[colback=black!3!white,colframe=black!30!white]
    \textbf{准则\thesection.\thethmCounter #1：}
}
{
    \end{tcolorbox}
}

\newenvironment{proposition}[1][]
{
    \refstepcounter{thmCounter}
    \begin{tcolorbox}[colback=yellow!10!white,colframe=black!30!white]
    \textbf{命题\thesection.\thethmCounter #1：}
}
{
    \end{tcolorbox}
}

% 证明和解
\newenvironment{proof}
{
    \begin{tcolorbox}[colback=black!3!white,colframe=black!30!white]
    \textbf{证明：}
}
{
    \end{tcolorbox}
}

\newenvironment{solution}
{
    \begin{tcolorbox}[colback=black!3!white,colframe=black!30!white]
    \textbf{解：}
}
{
    \end{tcolorbox}
}

\newenvironment{rmk}
{
    \begin{tcolorbox}[colback=black!3!white,colframe=black!30!white]
    \textbf{注：}
}
{
    \end{tcolorbox}
}

% 例
\newcounter{exampleCounter}[section]
\setcounter{exampleCounter}{0}
\newenvironment{example}[1][]
{
    \begin{tcolorbox}[colback=Emerald!20!white,colframe=black!30!white]
    \refstepcounter{exampleCounter}
    \textbf{例\thesection.\theexampleCounter #1：}
}
{
    \end{tcolorbox}
}
