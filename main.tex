% 定义
\newcounter{defnCounter}[section]
\setcounter{defnCounter}{0}
\newenvironment{defn}[1][]
{
    \refstepcounter{defnCounter}
    \begin{mdframed}[backgroundcolor=cyan!20]
    定义\thesection.\thedefnCounter #1：
}
{
    \end{mdframed}
}

% 定理、引理、推论、命题和准则
\newcounter{thmCounter}[section]
\setcounter{thmCounter}{0}
\newenvironment{thm}[1][]
{
    \refstepcounter{thmCounter}
    \begin{mdframed}[backgroundcolor=yellow!20]
    定理\thesection.\thethmCounter #1：
}
{
    \end{mdframed}
}

\newenvironment{lemma}[1][]
{
    \refstepcounter{thmCounter}
    \begin{mdframed}
    引理\thesection.\thethmCounter #1：
}
{
    \end{mdframed}
}

\newenvironment{corollary}[1][]
{
    \refstepcounter{thmCounter}
    \begin{mdframed}
    推论\thesection.\thethmCounter #1：
}
{
    \end{mdframed}
}

\newenvironment{criterion}[1][]
{
    \refstepcounter{thmCounter}
    \begin{mdframed}
    准则\thesection.\thethmCounter #1：
}
{
    \end{mdframed}
}

\newenvironment{proposition}[1][]
{
    \refstepcounter{thmCounter}
    \begin{mdframed}[backgroundcolor=yellow!20]
    命题\thesection.\thethmCounter #1：
}
{
    \end{mdframed}
}

% 证明和解
\newenvironment{proof}
{
    \begin{mdframed}
    证明：
}
{
    \end{mdframed}
}

\newenvironment{solution}
{
    \begin{mdframed}
    解：
}
{
    \end{mdframed}
}

\newenvironment{rmk}
{\begin{mdframed}
    注：
}
{
    \end{mdframed}
}

% 例
\newcounter{exampleCounter}[section]
\setcounter{exampleCounter}{0}
\newenvironment{example}[1][]
{
    \begin{mdframed}[backgroundcolor=Emerald!20]
    \refstepcounter{exampleCounter}
    例\thesection.\theexampleCounter #1：
}
{
    \end{mdframed}
}
